        "def check_first_wave_assignments(assignments: pd.DataFrame) -> None:\n",
        "    \"\"\"\n",
        "    Validates the results of the FIRST WAVE of student course assignments.\n",
        "\n",
        "    Parameters\n",
        "    ----------\n",
        "    assignments : pd.DataFrame\n",
        "        DataFrame with columns:\n",
        "            ['ID', 'course1', 'course2', 'fall_course_num']\n",
        "        One row = one student.\n",
        "        \n",
        "        - 'ID'               : student identifier (string)\n",
        "        - 'course1', 'course2': courses assigned in the first wave.\n",

        "        - 'fall_course_num'  : number of fall courses the student must choose (1 or 2)\n",

        "    Raises\n",
        "    ------\n",
        "    AssertionError\n",
        "        If any student's assignment does not match the reference (or allowed alternatives).\n",
        "\n",
        "    Notes\n",
        "    -----\n",
        "    Two specific students have multiple valid outcomes (ALTERNATIVES) due to ties in the algorithm.\n",
        "    The checker accepts any of the listed possibilities for them.\n",
        "    \"\"\"\n",
        "    \n",
        "    def course_columns_to_set_column(row: pd.Series) -> Counter:\n",
        "        \"\"\"Transform a row's course columns into a Counter of assigned courses.\"\"\"\n",
        "        result = Counter()\n",
        "        for col in ['course1', 'course2']:\n",
        "            val = row[col]\n",
        "            if pd.notna(val) and val is not None:\n",
        "                result[val] += 1\n",
        "        return result\n",
        "\n",
        "    ANSWER_SOURCE = './task_2_check.csv'\n",
        "\n",
        "    # Students with multiple acceptable assignments (due to algorithmic ties)\n",

        "        \"cdb87f7b15495e514eb481230b44540c6b0cfeb347c991f06b1e361fcdef0c0b\": [\n",
        "            Counter({\"Deep Learning for Text Data\", \"Bayesian Methods in Machine Learning\"}),\n",
        "            Counter({\"Deep Learning for Text Data\"}),\n",
        "        ],\n",
        "        \"6fa015d8846e21f4f79bac013b469a1e6e962fffd6121b4d9bcde3906275293e\": [\n",

        "        ],\n",
